import { NeonBadge } from '@/components/ui/neon-badge';
import { ProgressBar } from '@/components/ui/progress-bar';
import {
  ArrowLeft,
  Flame,
  Trophy,
  Star,
  Medal,
  Crown,
  Users,
  Skull,
  Globe,
  LogOut,
  Shield,
  Loader2,
  AlertCircle,
} from 'lucide-react';
import { cn } from '@/lib/utils';

const ProfileScreen: React.FC = () => {
  const {
    translate,
    setCurrentScreen,
    groups,
    language,
    setLanguage,
    profile,
    isAuthenticated,
    authLoading,
    authError,
    signInWithGoogle,
    signOut,
    authUser,
  } = useApp();

  // Loading state
  if (authLoading) {
    return (
      <div className="min-h-screen bg-background flex items-center justify-center">
        <div className="flex flex-col items-center gap-3">
          <Loader2 className="w-8 h-8 animate-spin text-primary" />

  }

  // Not authenticated - show sign in prompt

      <div className="min-h-screen bg-background flex flex-col">

          </GamerCard>

      </div>

          </GamerCard>

  const displayEmail = profile?.email || authUser?.email || '';
  const avatarUrl = profile?.avatar_url;
  const streak = profile?.streak || 0;

  const eventsAttended = profile?.events_attended || 0;
  const userBadges = profile?.badges || [];

                    onClick={signInWithGoogle}

                    className="gap-2"

                      <path fill="currentColor" d="M22.56 12.25c0-.78-.07-1.53-.2-2.25H12v4.26h5.92c-.26 1.37-1.04 2.53-2.21 3.31v2.77h3.57c2.08-1.92 3.28-4.74 3.28-8.09z"/>

            <GamerCardTitle>{translate('leaderboard')}</GamerCardTitle>

                    <div className="flex items-center gap-2">
                      <span className={cn(
                        "font-medium",
                        isCurrentUserEntry && "text-primary"
                      )}>
                        {user.name}
                        {isCurrentUserEntry && " (You)"}
                      </span>
                      {user.streak > 0 && (
                        <span className="text-xs text-orange-500">
                          ðŸ”¥{user.streak}

                      className="mt-1"

                  )}>
                    {user.rate}%

                </div>

            })}

          </GamerCard>

              </p>

              <div className="space-y-2">

                  <div 
                    key={group.id}
                    className="flex items-center justify-between p-3 rounded-lg bg-secondary/30 hover:bg-secondary/50 transition-colors cursor-pointer"

                      <div className="w-10 h-10 rounded-lg bg-gradient-gamer flex items-center justify-center text-lg">

                variant="neonOutline"
